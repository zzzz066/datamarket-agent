% --- 四个挑战 ---
\begin{frame}
	\frametitle{ML 训练数据为什么需要一个专门的交易市场}

	\begin{block}{ML 训练数据作为商品的四个独特挑战（Agarwal et al., 2019）}
		\begin{enumerate}
			\item \textbf{可无限复制}：零边际成本$\rightarrow$ 稀缺性定价模型直接失效
			\item \textbf{价值是组合性的}：一个数据集的价值取决于买方已有哪些其他数据$\rightarrow$ 无法独立定价
			\item \textbf{买方无法为单个数据集报价}：买方没有对特定数据集的先验价值判断
			\item \textbf{价值事前不可验证}：不先跑一遍 ML 任务，谁也不知道数据有没有用
		\end{enumerate}
	\end{block}

	\begin{alertblock}{关键}
		买方真正知道的不是``这个数据集值多少钱''，而是\textbf{``预测精度提升 1\% 对我来说值多少钱''}。\\
		因此市场的交易单位不应是``数据集''，而应是\textbf{预测精度的边际提升}。
	\end{alertblock}
\end{frame}

% --- 市场架构 ---
\begin{frame}
	\frametitle{论文提出的机制}

	\begin{columns}[T]
		\begin{column}{0.38\textwidth}
			\begin{block}{三方参与者}
				\begin{itemize}
					\item \textbf{平台}：中心化定价，运行 ML 模型，结算分成
					\item \textbf{买家}：提交预测任务 $Y_n$ 和报价 $b_n$
					\item \textbf{卖家}：提供数据特征 $X_j$，被动接受分成
				\end{itemize}
			\end{block}
			\vspace{0.3cm}
			\begin{center}
				\small\textbf{买方不接触原始数据，只拿到模型预测结果}
			\end{center}
		\end{column}
		\begin{column}{0.60\textwidth}
			\begin{block}{单轮交易}
				\begin{enumerate}
					\small
					\item 买家提交预测任务 $Y_n$
					\item 买家报价 $b_n$（\textit{``精度提升 1\% 付 \$X''}）
					\item AF 分配数据：$b_n \ge p_n$ 给完整数据，否则加噪声
					\item 平台训练 ML 模型，计算精度增益 $G(Y_n, \hat{Y}_n)$
					\item RF（Myerson）计算支付额：\textbf{诚实报价占优}
					\item PD（Shapley）分配收入给卖家：\textbf{复制者受罚}
					\item PF（MWU）更新下一轮价格：\textbf{无遗憾定价}
				\end{enumerate}
			\end{block}
		\end{column}
	\end{columns}
	\vspace{0.2cm}
	{\tiny Agarwal, Dahleh, and Sarkar. \textit{A Marketplace for Data: An Algorithmic Solution}. EC~\textquotesingle 19, 2019.}

\end{frame}

\begin{frame}
	\frametitle{四大机制及其设计理由}

	\begin{center}
		\footnotesize
		\renewcommand{\arraystretch}{1.6}
		\begin{tabular}{p{2.5cm}|p{2.5cm}|p{7cm}}
			\toprule
			\textbf{机制} & \textbf{应对的挑战} & \textbf{设计逻辑} \\
			\midrule
			Allocation Function（数据分配） & 事前不可验证 & 买方不接触原始数据：平台统一训练模型输出预测，避免数据泄露 \\
			\midrule
			Myerson Revenue Function（支付） & 策略性谎报 & Myerson 支付函数数学上保证 $b_n = \mu_n$ 是占优策略：诚实报价收益最高 \\

			\midrule
			Payment Division / Shapley（分配） & 可复制 + 组合价值 & Shapley 值衡量边际贡献；余弦相似度指数惩罚复制数据 \\
			\bottomrule
		\end{tabular}
	\end{center}

% --- 论文自身的局限性 ---

\begin{frame}
	\frametitle{该机制的局限性}

	\begin{block}{局限}
		\begin{enumerate}
			\item \textbf{买家是短视的}：只优化单轮效用，不考虑多轮策略
			\item \textbf{平台集中定价}：卖家不能自主定价，只能被动接受 Shapley 分成
			\item \textbf{基于严格的私有估值假设}：忽视了数据的竞争外部性（买家效用往往受对手是否获取数据的影响）
		\end{enumerate}
	\end{block}

	\vspace{0.2cm}
	\begin{center}
		\textbf{论文假定买家为理性、短视的私有估值者，诚实报价仅作为其在单阶段博弈下的占优策略}\\
	\end{center}
\end{frame}

\section{我们的方法}

\begin{frame}
	\frametitle{用 LLM Agent 模拟真实市场参与者}

	\begin{block}{核心思路}
		保持论文的\textbf{市场机制完全不变}，把规则化的买卖双方替换为 \textbf{LLM Agent}，观察行为偏差：
		\begin{itemize}
			\item 买家 Agent 面对价格决定报价策略
			\item 平台 Agent 根据市场反馈调整定价
			\item 对比理论最优 vs.\ LLM Agent 的行为差异
		\end{itemize}
	\end{block}

	\vspace{0.3cm}
	\begin{center}
		\textbf{核心方法}：四模式对比实验（all\_rule / seller\_agent / buyer\_agent / both\_agent）\\
		$\rightarrow$ 测量 Utility Gap $\rightarrow$ 定位机制偏差 $\rightarrow$ 提出改进
	\end{center}
\end{frame}
\begin{frame}
	\frametitle{核心研究问题}

	\begin{enumerate}
		\item \textbf{LLM Agent 的诚实性}
		      \begin{itemize}
			      \item 面对 Myerson 支付规则，LLM Agent 是否会自发学会诚实报价？
			      \item 诚实报价 vs.\ 策略性低报：Agent 选哪个？为什么？
		      \end{itemize}

		\pause
		\item \textbf{多轮博弈中的合谋行为}
		      \begin{itemize}
			      \item 多个买家 Agent 在反复交易中，是否会集体压低报价？
			      \item MWU 定价机制能否抵御这种隐性合谋导致的价格下行？
		      \end{itemize}

		\pause
		\item \textbf{机制的鲁棒性}
		      \begin{itemize}
			      \item 买卖双方都是 LLM Agent 博弈时，MWU 价格还会收敛吗？
			      \item Shapley 惩罚还能有效阻止复制行为吗？
		      \end{itemize}

		\pause
		\item \textbf{改进方向}
		      \begin{itemize}
			      \item 基于 Agent 的行为偏差，解释机制为什么有问题
			      \item 探索更贴近现实的合理市场机制设计
		      \end{itemize}
	\end{enumerate}
\end{frame}

\begin{frame}
	\frametitle{JSON 动作解析器：从自然语言到可执行动作}

	\begin{block}{解析目标}
		\small LLM 输出必须最终归一化为环境可执行的动作字典：平台 \texttt{\{"价格": p\}}，买家 \texttt{\{"报价": b\}}。
	\end{block}

	\vspace{0.2cm}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{支持的常见输出形态}
				\small
				\begin{itemize}
					\item 纯 JSON：\texttt{\{"报价": 0.8\}}
					\item Markdown JSON 代码块
					\item 先解释、后给 JSON
					\item 嵌套字段：\texttt{\{"action": \{"bid": 0.7\}\}}
					\item 英文字段别名：\texttt{price}, \texttt{bid}
				\end{itemize}
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{动作校验与统计}
				\small
				\begin{itemize}
					\item 平台价格裁剪到候选区间
					\item 买家报价裁剪为非负数
					\item 统计解析失败、fallback、越界动作
					\item 保存 \texttt{reasoning} 供后续行为分析
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Prompt 设计：角色感知 + 机制引导 + 短期记忆}

	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			\begin{block}{System Prompt 结构}
				\small
				\begin{enumerate}
					\item \textbf{角色目标}：平台最大化收益；买家最大化自身效用
					\item \textbf{信息边界}：平台不知道买家估值；买家不知道他人信息
					\item \textbf{机制说明}：AF 数据质量、RF 支付、PF/MWU 动态定价
					\item \textbf{实验 profile}：战略型、诚实型、福利型等
					\item \textbf{JSON schema}：强制输出可解析动作
				\end{enumerate}
			\end{block}
		\end{column}
		\begin{column}{0.46\textwidth}
			\begin{block}{User Prompt 每轮注入}
				\small
				\begin{itemize}
					\item 当前角色与 prompt profile
					\item 当前观察 \texttt{obs}
					\item 最近 $K$ 轮记忆
					\item 内部三步判断：趋势识别、效用影响、数字动作
				\end{itemize}
			\end{block}
			\begin{alertblock}{输出格式}
				\small \texttt{\{"reasoning": "一句话理由", "报价": 0.8\}} 或 \texttt{\{"reasoning": "一句话理由", "价格": 1.0\}}
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}


\begin{frame}
    \frametitle{Prompt 示例一：平台定价 Agent}

    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{System Prompt：\texttt{revenue\_max}}
                \scriptsize
                你是数据市场的平台定价 Agent。你只观察候选价格、MWU 推荐价格、卖家数量和最近交易历史；你不知道任何买家的私有估值。你的目标是在多轮交易中提高平台收入，同时避免价格过高导致交易失败。请基于历史反馈选择本轮价格。
            \end{exampleblock}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{exampleblock}{User Prompt：每轮注入}
                \scriptsize
                当前轮次、总轮次、候选价格、MWU 推荐价格、最近成交价/报价/平台收益。请先判断最近价格是否偏高或偏低，再从候选价格中选择一个价格，并输出 JSON：\texttt{\textbraceleft "reasoning":"...", "价格":1.0\textbraceright}。
            \end{exampleblock}
        \end{column}
    \end{columns}

    \vspace{0.2cm}
    \begin{block}{为什么这样设计？}
        \footnotesize 平台 prompt 刻意限制信息集，不让它读取买家真实估值；这样可以检验在与论文相同的不完全信息设定下，LLM 是否会偏离 MWU 推荐价、主动试探支付意愿，或因短期收益目标造成价格震荡。
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Prompt 示例二：买家 Agent 的三类报价策略}

    \begin{center}
        \scriptsize
        \renewcommand{\arraystretch}{1.35}
        \begin{tabular}{p{2.3cm}|p{4.0cm}|p{3.0cm}|p{3.1cm}}
            \toprule
            \textbf{Profile} & \textbf{Prompt 片段} & \textbf{设计目的} & \textbf{验证问题} \\
            \midrule
            \texttt{truthful} & 你的目标是最大化单轮效用；Myerson 支付规则下，诚实报价 $b=\mu$ 是占优策略。 & 理论基线 & LLM 能否按机制理论诚实报价？ \\
            \midrule
            \texttt{strategic} & 除本轮效用外，也考虑你的报价是否会影响未来价格路径；不要编造不可见信息。 & 多轮策略 & LLM 会不会为压低未来价格而低报？ \\
            \midrule
            \texttt{shade} & 如果当前价格低于估值，可以考虑适度低报以节省支付，但要权衡拿不到高质量数据的风险。 & 低报压力测试 & Myerson+MWU 能否抵御策略性低报？ \\
            \bottomrule
        \end{tabular}
    \end{center}

    \vspace{0.15cm}
    \begin{alertblock}{统一输出}
        \footnotesize \texttt{\textbraceleft "reasoning":"一句话说明诚实/低报/高报原因", "报价":0.8\textbraceright}；保留 reasoning 是为了把数值偏离和 Agent 的主观解释对应起来。
    \end{alertblock}
\end{frame}

\begin{frame}
    \frametitle{Prompt 示例三：竞争外部性与风险偏好}

    \begin{columns}[T]
        \begin{column}{0.49\textwidth}
            \begin{exampleblock}{竞争型买家：\texttt{competitive}}
                \scriptsize
                你与另一个买家在下游市场竞争。如果对手获得高质量预测而你没有获得，你的商业收益会下降；如果双方都获得数据，预测优势会被部分抵消。请在当前价格、你的估值、对手可能购买的风险之间选择报价。
            \end{exampleblock}
            \vspace{0.12cm}
            \begin{block}{验证问题}
                \scriptsize 私有估值假设被打破后，买家估值是否会随对手行为动态变化？诚实报价结论是否仍然稳定？
            \end{block}
        \end{column}
        \begin{column}{0.49\textwidth}
            \begin{exampleblock}{风险规避买家：\texttt{risk\_averse}}
                \scriptsize
                你的预算有限，负效用会带来额外风险成本。若价格接近估值，请偏向保守报价；若数据质量提升能显著降低业务风险，可以接受略高报价。输出时说明你主要规避的是支付风险还是质量风险。
            \end{exampleblock}
            \vspace{0.12cm}
            \begin{block}{验证问题}
                \scriptsize 当参与者不是风险中性时，原机制的收益、效率和成交率是否出现系统性变化？
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}
    \frametitle{Prompt 设计动机与可验证问题}

    \begin{center}
        \scriptsize
        \renewcommand{\arraystretch}{1.4}
        \begin{tabular}{p{2.5cm}|p{4.0cm}|p{5.9cm}}
            \toprule
            \textbf{设计元素} & \textbf{为什么这样设计} & \textbf{能验证什么问题} \\
            \midrule
            角色目标 & 平台、买家优化目标不同，避免所有 Agent 说出同一种“理性”答案 & 平台收益最大化与买家效用最大化是否会产生机制冲突 \\
            \midrule
            信息边界 & 明确“不可见估值/他人历史”，对齐论文的不完全信息假设 & LLM 是否会在信息受限下仍然推断、臆测或策略性行动 \\
            \midrule
            机制说明 & 告诉 Agent AF/RF/PF/PD 规则，但不直接给最优动作 & 机制知识是否足以诱导诚实报价或稳定定价 \\
            \midrule
            Profile 切换 & 在同一市场环境下只改变 prompt 偏好 & 行为偏差来自市场机制，还是来自 Agent 偏好设定 \\
            \midrule
            JSON+reasoning & 动作可执行，理由可分析 & 数值异常、fallback、低报/高报能否被解释和统计 \\
            \bottomrule
        \end{tabular}
    \end{center}
\end{frame}
\begin{frame}
	\frametitle{Prompt Profiles：用于行为对照的 Agent 类型}

	\begin{center}
		\footnotesize
		\renewcommand{\arraystretch}{1.35}
		\begin{tabular}{p{2.7cm}|p{2.2cm}|p{8.0cm}}
			\toprule
			\textbf{Profile} & \textbf{角色} & \textbf{行为假设} \\
			\midrule
			\texttt{truthful} & 买家 & 强调诚实报价 $b=\mu$，作为理论基线 \\
			\texttt{strategic} & 买家/平台 & 允许考虑当前动作对未来价格路径的影响 \\
			\texttt{shade} & 买家 & 允许适度低报，观察 MWU 是否被压价 \\
			\texttt{revenue\_max} & 平台 & 更强调平台收入和平台效用 \\
			\texttt{fairness\_aware} & 平台 & 同时关注买家效用和卖家参与激励 \\
			\texttt{welfare}/\texttt{risk\_averse} & 双方 & 分别强调社会福利或规避负效用风险 \\
			\bottomrule
		\end{tabular}
	\end{center}

	\vspace{0.25cm}
	\begin{block}{中期实验用途}
		\small 通过固定市场环境、切换 prompt profile，比较不同 Agent 行为对平台收益、买家效用、卖家收益和动作合法性的影响。
	\end{block}
\end{frame}

\section{模拟实验及结果}

\begin{frame}
  \frametitle{实验流程}
  \begin{block}{模块定位}
    负责把市场机制环境与外部 Agent 接口组织成可重复运行的多轮实验。
  \end{block}

  \vspace{0.25cm}
  交易轮数由买家序列长度决定：
  \[
    \texttt{total\_rounds}=\texttt{len(buyers)}
  \]

  \vspace{0.25cm}
  一个 episode 表示一段连续交易过程：
  \begin{itemize}
    \item 卖家数据池在实验开始前固定；
    \item 买家按照预设序列依次进入市场；
    \item 每一轮中，平台先定价，当前买家再报价；
    \item 底层 AF/RF/PD/PF 完成结算与价格状态更新；
    \item 实验系统记录该轮结果并进入下一轮。
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{四种实验模式}
  为比较规则机制与 Agent 决策的影响，实验设计四种模式。

  \vspace{0.25cm}
  \begin{center}
    \begin{tabular}{>{\centering\arraybackslash}p{2.8cm}
                    >{\centering\arraybackslash}p{3.0cm}
                    >{\centering\arraybackslash}p{2.6cm}
                    p{4.0cm}}
      \toprule
      模式 & 平台/卖方定价侧 & 买家侧 & 实验目的 \\
      \midrule
      \texttt{all\_rule} & MWU 规则 & 诚实报价规则 & 作为规则基准组 \\
      \texttt{seller\_agent} & Agent & 诚实报价规则 & 测试平台 Agent 定价影响 \\
      \texttt{buyer\_agent} & MWU 规则 & Agent & 测试买家 Agent 报价影响 \\
      \texttt{both\_agent} & Agent & Agent & 测试双边 Agent 交互效果 \\
      \bottomrule
    \end{tabular}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{信息控制}
  实验提供统一接口：
  \[
    \texttt{agent.act(obs)} \rightarrow \texttt{action}
  \]

  \vspace{0.25cm}
  观测信息按角色构造：
  \begin{itemize}
    \item 平台侧可见当前轮次、总轮次、候选价格、卖家数量、最近历史；
    \item 买家侧可见当前轮次、总轮次、自身估值 $\mu$、当前价格、自身历史；
  \end{itemize}

  \vspace{0.25cm}
  当前可被 Agent 接管的两个决策点：
  \begin{itemize}
    \item 平台侧定价：\texttt{\{"价格": p\}}
    \item 买家侧报价：\texttt{\{"报价": b\}}
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{实验记录与模块输出}
  每次外部 Agent 决策都会记录一条 step 日志：
  \[
    \{\texttt{role},\ \texttt{observation},\ \texttt{action},\ \texttt{reward},\ \texttt{reasoning}\}
  \]

  \vspace{0.25cm}
  episode 结束后输出 \texttt{EpisodeRunResult}：
  \begin{itemize}
    \item \texttt{steps}：逐轮日志，包含 \texttt{raw\_output}、\texttt{reasoning}、\texttt{fallback\_used}；
    \item \texttt{summary}：平台总效用、买家平均效用、卖家效用等指标；
    \item \texttt{action\_stats}：fallback 次数、非法动作次数、缺少 reasoning 次数；
    \item 自动保存 \texttt{baseline\_results.json}、\texttt{summary\_table.csv} 和 \texttt{step\_table.csv}。
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{分析目标}
  \begin{block}{核心问题}
    Agent 接入后，市场结果相对规则基准组发生了怎样的变化？
  \end{block}

  \vspace{0.3cm}
  数据分析主要比较：
  \begin{itemize}
    \item 不同模式下的平台总效用；
    \item 不同模式下的买家平均效用；
    \item Agent 是否改变价格、报价、数据增益和平台收入的动态轨迹。
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{基准组与指标定义}
  以下分析以 \texttt{all\_rule} 作为基准组：
  \[
    \texttt{all\_rule}=\text{MWU 定价}+\text{买家诚实报价}
  \]

  \vspace{0.3cm}
  Utility Gap 定义为：
  \[
    \text{Utility Gap}=\text{规则基准组效用}-\text{当前模式效用}
  \]

  Pass@epsilon 用于判断某个模式是否足够接近规则基准组：
  \[
    |\texttt{platform\_gap}| \leq \epsilon,\quad
    |\texttt{buyer\_average\_gap}| \leq \epsilon
  \]
\end{frame}

	\begin{enumerate}
