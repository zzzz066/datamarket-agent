% --- 四个挑战 ---
\begin{frame}
	\frametitle{ML 训练数据为什么需要一个专门的交易市场}

	\begin{block}{ML 训练数据作为商品的四个独特挑战（Agarwal et al., 2019）}
		\begin{enumerate}
			\item \textbf{可无限复制}：零边际成本$\rightarrow$ 稀缺性定价模型直接失效
			\item \textbf{价值是组合性的}：一个数据集的价值取决于买方已有哪些其他数据$\rightarrow$ 无法独立定价
			\item \textbf{买方无法为单个数据集报价}：买方没有对特定数据集的先验价值判断
			\item \textbf{价值事前不可验证}：不先跑一遍 ML 任务，谁也不知道数据有没有用
		\end{enumerate}
	\end{block}

	\begin{alertblock}{关键}
		买方真正知道的不是``这个数据集值多少钱''，而是\textbf{``预测精度提升 1\% 对我来说值多少钱''}。\\
		因此市场的交易单位不应是``数据集''，而应是\textbf{预测精度的边际提升}。
	\end{alertblock}
\end{frame}

% --- 市场架构 ---
\begin{frame}
	\frametitle{论文提出的机制}

	\begin{columns}[T]
		\begin{column}{0.38\textwidth}
			\begin{block}{三方参与者}
				\begin{itemize}
					\item \textbf{平台}：中心化定价，运行 ML 模型，结算分成
					\item \textbf{买家}：提交预测任务 $Y_n$ 和报价 $b_n$
					\item \textbf{卖家}：提供数据特征 $X_j$，被动接受分成
				\end{itemize}
			\end{block}
			\vspace{0.3cm}
			\begin{center}
				\small\textbf{买方不接触原始数据，只拿到模型预测结果}
			\end{center}
		\end{column}
		\begin{column}{0.60\textwidth}
			\begin{block}{单轮交易}
				\begin{enumerate}
					\small
					\item 买家提交预测任务 $Y_n$
					\item 买家报价 $b_n$（\textit{``精度提升 1\% 付 \$X''}）
					\item AF 分配数据：$b_n \ge p_n$ 给完整数据，否则加噪声
					\item 平台训练 ML 模型，计算精度增益 $G(Y_n, \hat{Y}_n)$
					\item RF（Myerson）计算支付额：\textbf{诚实报价占优}
					\item PD（Shapley）分配收入给卖家：\textbf{复制者受罚}
					\item PF（MWU）更新下一轮价格：\textbf{无遗憾定价}
				\end{enumerate}
			\end{block}
		\end{column}
	\end{columns}
	\vspace{0.2cm}
	{\tiny Agarwal, Dahleh, and Sarkar. \textit{A Marketplace for Data: An Algorithmic Solution}. EC~\textquotesingle 19, 2019.}

\end{frame}

% --- 论文自身的局限性 ---

\begin{frame}
	\frametitle{该机制的局限性}

	\begin{block}{局限}
		\begin{enumerate}
			\item \textbf{买家是短视的}：只优化单轮效用，不考虑多轮策略
			\item \textbf{平台集中定价}：卖家不能自主定价，只能被动接受 Shapley 分成
			\item \textbf{基于严格的私有估值假设}：忽视了数据的竞争外部性（买家效用往往受对手是否获取数据的影响）
		\end{enumerate}
	\end{block}

	\vspace{0.2cm}
	\begin{center}
		\textbf{论文假定买家为理性、短视的私有估值者，诚实报价仅作为其在单阶段博弈下的占优策略}\\
	\end{center}
\end{frame}

\section{我们的方法}

\begin{frame}
	\frametitle{用 LLM Agent 模拟真实市场参与者}

	\begin{block}{核心思路}
		保持论文的\textbf{市场机制完全不变}（AF / RF / PF / PD 照原样实现），\\
		把规则化的买卖双方替换为 \textbf{LLM Agent}，观察行为偏差。
	\end{block}

	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{买家 Agent（LLM）}
				\textbf{看到}：当前价格 $p_n$、数据描述、历史成交、自己的估值 $\mu_n$、预算\\
				\textbf{决策}：报价 $b_n$ + 决策理由\\
				\textbf{研究}：Agent 会诚实报价吗？什么因素导致偏离？
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{卖家 Agent（LLM）}
				\textbf{看到}：自己的数据特征、历史分成、市场行情\\
				\textbf{决策}：是否复制数据、是否调整数据质量\\
				\textbf{研究}：Shapley 惩罚能否阻止 Agent 的复制行为？
			\end{exampleblock}
		\end{column}
	\end{columns}

	\pause
	\vspace{0.3cm}
	\begin{center}
		\textbf{对比实验}：理论最优vs.\ LLM Agent $\rightarrow$ 测量行为偏差 $\rightarrow$ 提出机制改进
	\end{center}
\end{frame}

\begin{frame}
	\frametitle{核心研究问题}

	\begin{enumerate}
		\item \textbf{LLM Agent 的诚实性}
		      \begin{itemize}
			      \item 面对 Myerson 支付规则，LLM Agent 是否会自发学会诚实报价？
			      \item 诚实报价 vs.\ 策略性低报：Agent 选哪个？为什么？
		      \end{itemize}

		\pause
		\item \textbf{多轮博弈中的合谋行为}
		      \begin{itemize}
			      \item 多个买家 Agent 在反复交易中，是否会集体压低报价？
			      \item MWU 定价机制能否抵御这种隐性合谋导致的价格下行？
		      \end{itemize}

		\pause
		\item \textbf{机制的鲁棒性}
		      \begin{itemize}
			      \item 买卖双方都是 LLM Agent 博弈时，MWU 价格还会收敛吗？
			      \item Shapley 惩罚还能有效阻止复制行为吗？
		      \end{itemize}

		\pause
		\item \textbf{改进方向}
		      \begin{itemize}
			      \item 基于 Agent 的行为偏差，解释机制为什么有问题
			      \item 探索更贴近现实的合理市场机制设计
		      \end{itemize}
		
		\pause
		\item \textbf{私有估值假设与竞争外部性}
		      \begin{itemize}
			      \item 当引入商业竞争（对手购买同一数据）时，Agent 的估值是否会发生动态耦合？
			      \item 在存在竞争外部性（非私有估值）时，原本保证诚实的 Myerson 机制是否会面临失效？
		      \end{itemize}
	\end{enumerate}
\end{frame}

	\vspace{0.35cm}
	\begin{columns}[T]
		\begin{column}{0.49\textwidth}
			\begin{block}{统一接口}
				\small
				所有参与者继承 \texttt{BaseAgent}：
				\[
					\texttt{act(obs)} \rightarrow \texttt{action}
				\]
				
			\end{block}
		\end{column}
		\begin{column}{0.49\textwidth}
			\begin{block}{实现模块}
				\small
				\texttt{LLMAgent} 负责构造 prompt 与调用模型；\texttt{OpenAICompatibleClient} 负责 HTTP API；\texttt{ActionParser} 负责 JSON 动作解析。
			\end{block}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{LLM 接口对接：OpenAI-compatible Chat API}

	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{block}{API 配置}
				\small
				通过环境变量或命令行接入第三方兼容接口：
				\begin{itemize}
					\item \texttt{OPENAI\_API\_KEY}
					\item \texttt{OPENAI\_BASE\_URL}
					\item \texttt{--model gpt-5.4}
				\end{itemize}
				
			\end{block}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{block}{工程鲁棒性}
				\small
				\begin{itemize}
					\item 支持不同兼容服务的 \texttt{base\_url}
					\item 默认要求 JSON object 输出
					\item 支持失败重试与 fallback 动作
					\item 日志记录 raw output 与错误原因
				\end{itemize}
			\end{block}
		\end{column}
	\end{columns}

	\vspace{0.35cm}
	\begin{alertblock}{为什么需要 fallback?}
		\small LLM 可能出现模型名不支持、JSON mode 不兼容、输出格式偏离等问题；fallback 保证一次坏输出不会中断整个 episode。
	\end{alertblock}
\end{frame}

\begin{frame}
	\frametitle{JSON 动作解析器：从自然语言到可执行动作}

	\begin{block}{解析目标}
		\small LLM 输出必须最终归一化为环境可执行的动作字典：平台 \texttt{\{"价格": p\}}，买家 \texttt{\{"报价": b\}}。
	\end{block}

	\vspace{0.2cm}
	\begin{columns}[T]
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{支持的常见输出形态}
				\small
				\begin{itemize}
					\item 纯 JSON：\texttt{\{"报价": 0.8\}}
					\item Markdown JSON 代码块
					\item 先解释、后给 JSON
					\item 嵌套字段：\texttt{\{"action": \{"bid": 0.7\}\}}
					\item 英文字段别名：\texttt{price}, \texttt{bid}
				\end{itemize}
			\end{exampleblock}
		\end{column}
		\begin{column}{0.48\textwidth}
			\begin{exampleblock}{动作校验与统计}
				\small
				\begin{itemize}
					\item 平台价格裁剪到候选区间
					\item 买家报价裁剪为非负数
					\item 统计解析失败、fallback、越界动作
					\item 保存 \texttt{reasoning} 供后续行为分析
				\end{itemize}
			\end{exampleblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
	\frametitle{Prompt 设计：角色感知 + 机制引导 + 短期记忆}

	\begin{columns}[T]
		\begin{column}{0.50\textwidth}
			\begin{block}{System Prompt 结构}
				\small
				\begin{enumerate}
					\item \textbf{角色目标}：平台最大化收益；买家最大化自身效用
					\item \textbf{信息边界}：平台不知道买家估值；买家不知道他人信息
					\item \textbf{机制说明}：AF 数据质量、RF 支付、PF/MWU 动态定价
					\item \textbf{实验 profile}：战略型、诚实型、福利型等
					\item \textbf{JSON schema}：强制输出可解析动作
				\end{enumerate}
			\end{block}
		\end{column}
		\begin{column}{0.46\textwidth}
			\begin{block}{User Prompt 每轮注入}
				\small
				\begin{itemize}
					\item 当前角色与 prompt profile
					\item 当前观察 \texttt{obs}
					\item 最近 $K$ 轮记忆
					\item 内部三步判断：趋势识别、效用影响、数字动作
				\end{itemize}
			\end{block}
			\begin{alertblock}{输出格式}
				\small \texttt{\{"reasoning": "一句话理由", "报价": 0.8\}} 或 \texttt{\{"reasoning": "一句话理由", "价格": 1.0\}}
			\end{alertblock}
		\end{column}
	\end{columns}
\end{frame}

\begin{frame}
    \frametitle{Prompt 示例：平台与买家 Agent}

    \begin{columns}[T]
        \begin{column}{0.49\textwidth}
            \begin{exampleblock}{平台 Agent：\texttt{revenue\_max}}
                \footnotesize
                \textbf{System:} 你是数据市场的平台定价 Agent。可见信息只有候选价格、MWU 推荐价格、卖家数量和最近历史；不要假设买家的私有估值。目标是在多轮交易中提高平台效用和收入。

                \textbf{User:} 当前角色=平台；profile=\texttt{revenue\_max}；观察包含候选价格、MWU 推荐价格、最近历史。比较 MWU 推荐价、最近报价和平台收入后选择价格。

                \textbf{Output:} \texttt{\textbraceleft "reasoning":"提高价格试探支付意愿", "价格":1.0\textbraceright}
            \end{exampleblock}
        \end{column}
        \begin{column}{0.49\textwidth}
            \begin{exampleblock}{买家 Agent：\texttt{strategic}}
                \footnotesize
                \textbf{System:} 你是数据市场的买家 Agent。你只知道自己的估值 $\mu$、当前平台价格和自己的历史；目标是最大化自身效用：估值 $\times$ 数据质量 $-$ 支付。

                \textbf{User:} 当前角色=买家；profile=\texttt{strategic}；观察包含当前价格、我的估值、我的历史。比较诚实报价、低报节省支付和高报获得完整数据的风险。

                \textbf{Output:} \texttt{\textbraceleft "reasoning":"价格低于估值，诚实报价可获得高质量数据", "报价":0.8\textbraceright}
            \end{exampleblock}
        \end{column}
    \end{columns}

    \begin{alertblock}{设计原则}
        \footnotesize Prompt 中明确角色目标、信息边界、机制规则和 JSON schema；reasoning 只保留一句话，便于后续行为解释。
    \end{alertblock}
\end{frame}
\begin{frame}
	\frametitle{Prompt Profiles：用于行为对照的 Agent 类型}

	\begin{center}
		\footnotesize
		\renewcommand{\arraystretch}{1.35}
		\begin{tabular}{p{2.7cm}|p{2.2cm}|p{8.0cm}}
			\toprule
			\textbf{Profile} & \textbf{角色} & \textbf{行为假设} \\
			\midrule
			\texttt{truthful} & 买家 & 强调诚实报价 $b=\mu$，作为理论基线 \\
			\texttt{strategic} & 买家/平台 & 允许考虑当前动作对未来价格路径的影响 \\
			\texttt{shade} & 买家 & 允许适度低报，观察 MWU 是否被压价 \\
			\texttt{revenue\_max} & 平台 & 更强调平台收入和平台效用 \\
			\texttt{fairness\_aware} & 平台 & 同时关注买家效用和卖家参与激励 \\
			\texttt{welfare}/\texttt{risk\_averse} & 双方 & 分别强调社会福利或规避负效用风险 \\
			\bottomrule
		\end{tabular}
	\end{center}

	\vspace{0.25cm}
	\begin{block}{中期实验用途}
		\small 通过固定市场环境、切换 prompt profile，比较不同 Agent 行为对平台收益、买家效用、卖家收益和动作合法性的影响。
	\end{block}
\end{frame}

\section{模拟实验及结果}

\begin{frame}
  \frametitle{实验流程}
  \begin{block}{模块定位}
    负责把市场机制环境与外部 Agent 接口组织成可重复运行的多轮实验。
  \end{block}

  \vspace{0.25cm}
  交易轮数由买家序列长度决定：
  \[
    \texttt{total\_rounds}=\texttt{len(buyers)}
  \]

  \vspace{0.25cm}
  一个 episode 表示一段连续交易过程：
  \begin{itemize}
    \item 卖家数据池在实验开始前固定；
    \item 买家按照预设序列依次进入市场；
    \item 每一轮中，平台先定价，当前买家再报价；
    \item 底层 AF/RF/PD/PF 完成结算与价格状态更新；
    \item 实验系统记录该轮结果并进入下一轮。
  \end{itemize}

\end{frame}

\begin{frame}
  \frametitle{四种实验模式}
  为比较规则机制与 Agent 决策的影响，实验设计四种模式。

  \vspace{0.25cm}
  \begin{center}
    \begin{tabular}{>{\centering\arraybackslash}p{2.8cm}
                    >{\centering\arraybackslash}p{3.0cm}
                    >{\centering\arraybackslash}p{2.6cm}
                    p{4.0cm}}
      \toprule
      模式 & 平台/卖方定价侧 & 买家侧 & 实验目的 \\
      \midrule
      \texttt{all\_rule} & MWU 规则 & 诚实报价规则 & 作为规则基准组 \\
      \texttt{seller\_agent} & Agent & 诚实报价规则 & 测试平台 Agent 定价影响 \\
      \texttt{buyer\_agent} & MWU 规则 & Agent & 测试买家 Agent 报价影响 \\
      \texttt{both\_agent} & Agent & Agent & 测试双边 Agent 交互效果 \\
      \bottomrule
    \end{tabular}
  \end{center}

\end{frame}

\begin{frame}
  \frametitle{信息控制}
  实验提供统一接口：
  \[
    \texttt{agent.act(obs)} \rightarrow \texttt{action}
  \]

  \vspace{0.25cm}
  观测信息按角色构造：
  \begin{itemize}
    \item 平台侧可见当前轮次、总轮次、候选价格、卖家数量、最近历史；
    \item 买家侧可见当前轮次、总轮次、自身估值 $\mu$、当前价格、自身历史；
  \end{itemize}

  \vspace{0.25cm}
  当前可被 Agent 接管的两个决策点：
  \begin{itemize}
    \item 平台侧定价：\texttt{\{"价格": p\}}
    \item 买家侧报价：\texttt{\{"报价": b\}}
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{实验记录与模块输出}
  每次外部 Agent 决策都会记录一条 step 日志：
  \[
    \{\texttt{role},\ \texttt{observation},\ \texttt{action},\ \texttt{reward},\ \texttt{reasoning}\}
  \]

  \vspace{0.25cm}
  episode 结束后输出 \texttt{EpisodeRunResult}：
  \begin{itemize}
    \item \texttt{steps}：逐轮日志，包含 \texttt{raw\_output}、\texttt{reasoning}、\texttt{fallback\_used}；
    \item \texttt{summary}：平台总效用、买家平均效用、卖家效用等指标；
    \item \texttt{action\_stats}：fallback 次数、非法动作次数、缺少 reasoning 次数；
    \item 自动保存 \texttt{baseline\_results.json}、\texttt{summary\_table.csv} 和 \texttt{step\_table.csv}。
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{分析目标}
  \begin{block}{核心问题}
    Agent 接入后，市场结果相对规则基准组发生了怎样的变化？
  \end{block}

  \vspace{0.3cm}
  数据分析主要比较：
  \begin{itemize}
    \item 不同模式下的平台总效用；
    \item 不同模式下的买家平均效用；
    \item Agent 是否改变价格、报价、数据增益和平台收入的动态轨迹。
  \end{itemize}
\end{frame}

\begin{frame}
	\frametitle{基准组与指标定义}
  以下分析以 \texttt{all\_rule} 作为基准组：
  \[
    \texttt{all\_rule}=\text{MWU 定价}+\text{买家诚实报价}
  \]

  \vspace{0.3cm}
  Utility Gap 定义为：
  \[
    \text{Utility Gap}=\text{规则基准组效用}-\text{当前模式效用}
  \]

  Pass@epsilon 用于判断某个模式是否足够接近规则基准组：
  \[
    |\texttt{platform\_gap}| \leq \epsilon,\quad
    |\texttt{buyer\_average\_gap}| \leq \epsilon
  \]
\end{frame}
